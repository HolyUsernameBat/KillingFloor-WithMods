function bool CanAttack(Actor Other)
{

  local float enemyDist;
  local Bot B;

  if(Bot(Instigator.Controller) != none)
  {
    B = Bot(Instigator.Controller);
    enemyDist = VSize(B.Enemy.Location - Instigator.Location);

    if (enemyDist < 110 )
    {
        return true;
    }
    else
      return false;
  }
  return true ;
}
